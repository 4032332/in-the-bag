# Explore Screen Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement the Explore tab with two AI-powered modes — Find a Holiday (conversational destination suggestions) and Enhance My Trip (YouTube/TikTok URL extraction to trip events and tasks) — gated behind a premium paywall for free users.

**Architecture:** The Explore tab is a standalone tab screen with two sub-modes toggled via a segmented control; each mode is a separate component that communicates with Supabase Edge Functions via async jobs, listening for completion via Supabase Realtime on the `async_jobs` table. Find a Holiday uses the `ai_trip_suggest` Edge Function and stores conversation history exclusively in MMKV local storage (never synced to Supabase); Enhance My Trip queues `youtube_extract` or `tiktok_extract` jobs and presents results as a user-controlled checklist before writing to `trip_tasks` or `events`.

**Tech Stack:** Expo Router (tab navigation), Supabase JS client (direct writes + Realtime subscription), MMKV (conversation history), React Native Reanimated (loading states, toast), Zod (job output validation)

## Task 2 — MMKV Conversation History Storage

**Goal:** Define and implement the conversation history storage contract so FindHoliday can persist and restore chat history locally.

**Storage schema:**

- MMKV key: `explore_conversation_history_${userId}` (user-scoped to prevent cross-account leakage on shared devices)
- Value: JSON-serialised `ConversationMessage[]`
- The array is the full ordered conversation from oldest to newest
- Cleared (key deleted) when the user taps "New Conversation"
- Never written to or read from Supabase — local-only

- [ ] Create `components/explore/FindHoliday/conversationStorage.ts`

```ts
// components/explore/FindHoliday/conversationStorage.ts
import { storage } from '@/lib/mmkv';
import { ConversationMessage } from '@/types/explore';

// MI-10: Key is prefixed with userId to prevent cross-account leakage on shared devices.
// Pass the authenticated user's ID (from supabase.auth.getUser()) as the userId parameter.
function conversationKey(userId: string): string {
  return `explore_conversation_history_${userId}`;
}

export function loadConversation(userId: string): ConversationMessage[] {
  const raw = storage.getString(conversationKey(userId));
  if (!raw) return [];
  try {
    return JSON.parse(raw) as ConversationMessage[];
  } catch {
    return [];
  }
}

export function saveConversation(userId: string, messages: ConversationMessage[]): void {
  storage.set(conversationKey(userId), JSON.stringify(messages));
}

export function clearConversation(userId: string): void {
  storage.delete(conversationKey(userId));
}
```

## Task 4 — Find a Holiday: Screen Logic and Edge Function Integration

**Goal:** Wire up the full FindHoliday conversation flow: load history from MMKV, send messages to `ai_trip_suggest` via an async job, receive results via Realtime, persist history, and handle "New Conversation".

- [ ] Create `hooks/useAsyncJob.ts`
  - Subscribes to `async_jobs` table via Supabase Realtime, filtered to a specific `job_id`
  - Returns `{ status, output, error }`
  - Unsubscribes on unmount or when `job_id` changes to `null`

```ts
// hooks/useAsyncJob.ts
import { useEffect, useState } from 'react';
import { supabase } from '@/lib/supabase';

type JobStatus = 'pending' | 'processing' | 'completed' | 'failed' | null;

interface AsyncJobState {
  status: JobStatus;
  output: Record<string, unknown> | null;
  error: string | null;
}

export function useAsyncJob(jobId: string | null): AsyncJobState {
  const [state, setState] = useState<AsyncJobState>({ status: null, output: null, error: null });

  useEffect(() => {
    if (!jobId) {
      setState({ status: null, output: null, error: null });
      return;
    }

    // Fetch initial state
    supabase
      .from('async_jobs')
      .select('status, output, error')
      .eq('id', jobId)
      .single()
      .then(({ data }) => {
        if (data) setState({ status: data.status, output: data.output, error: data.error });
      });

    // Subscribe to updates
    const channel = supabase
      .channel(`async_job_${jobId}`)
      .on(
        'postgres_changes',
        { event: 'UPDATE', schema: 'public', table: 'async_jobs', filter: `id=eq.${jobId}` },
        (payload) => {
          const { status, output, error } = payload.new as {
            status: JobStatus;
            output: Record<string, unknown> | null;
            error: string | null;
          };
          setState({ status, output, error });
        }
      )
      .subscribe();

    return () => {
      supabase.removeChannel(channel);
    };
  }, [jobId]);

  return state;
}
```

export default function FindHolidayScreen() {
  const [messages, setMessages] = useState<ConversationMessage[]>(() => loadConversation());
  const [loading, setLoading] = useState(false);
  const listRef = useRef<FlatList>(null);
  const router = useRouter();

  const scrollToBottom = () => {
    setTimeout(() => listRef.current?.scrollToEnd({ animated: true }), 100);
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  const handleNewConversation = useCallback(() => {
    clearConversation();
    setMessages([]);
  }, []);

  const handleSend = useCallback(async (text: string) => {
    const userMessage: ConversationMessage = {
      id: uuidv4(),
      role: 'user',
      content: text,
      timestamp: Date.now(),
    };

    const updatedMessages = [...messages, userMessage];
    setMessages(updatedMessages);
    saveConversation(updatedMessages);
    setLoading(true);

    try {
      // Build context payload — exclude suggestionCards from history (not needed by backend)
      const historyForBackend = updatedMessages.map(({ role, content }) => ({ role, content }));

      const { data, error } = await supabase.functions.invoke('ai_trip_suggest', {
        body: { message: text, conversationHistory: historyForBackend },
      });

      if (error) throw error;

      const { reply, suggestionCards } = data as {
        reply: string;
        suggestionCards?: SuggestionCard[];
      };

        id: uuidv4(),
        role: 'assistant',
        content: reply,
        timestamp: Date.now(),
        suggestionCards: suggestionCards ?? [],
      };

      const withAssistant = [...updatedMessages, assistantMessage];
      setMessages(withAssistant);
      saveConversation(withAssistant);
    } catch (err) {
      const errorMessage: ConversationMessage = {
        id: uuidv4(),
        role: 'assistant',
        content: 'Something went wrong. Please try again.',
        timestamp: Date.now(),
      };
      const withError = [...updatedMessages, errorMessage];
      setMessages(withError);
      saveConversation(withError);
    } finally {
      setLoading(false);
    }
  }, [messages]);

  const handleCardTap = useCallback((card: SuggestionCard) => {

  }, [router]);

  return (

      </View>

        <View style={styles.emptyState}>
          <Text style={styles.emptyText}>
            Describe what kind of holiday you are looking for and get personalised destination suggestions.
          </Text>
        </View>
      )}

      <FlatList
        ref={listRef}
        data={messages}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <ChatBubble message={item} onCardTap={handleCardTap} />
        )}
        contentContainerStyle={styles.listContent}
        onLayout={scrollToBottom}
      />

      {loading && (
        <View style={styles.typingIndicator}>
          <ActivityIndicator size="small" color="#007AFF" />
          <Text style={styles.typingText}>Thinking...</Text>
        </View>
      )}

      <ChatInput onSend={handleSend} disabled={loading} />
    </View>
  );
}

import { useEffect, useState } from 'react';
import { supabase } from '@/lib/supabase';

export interface TripSummary {
  id: string;
  name: string;
  destinations: Array<{ city: string; country: string; start_date: string; end_date: string }>;
}

export function useTripList() {
  const [trips, setTrips] = useState<TripSummary[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    supabase
      .from('trips')
      .select('id, name, trip_destinations(city, country, start_date, end_date)')
      .order('created_at', { ascending: false })
      .then(({ data }) => {
        if (data) {
          setTrips(
            data.map((t) => ({
              id: t.id,
              name: t.name,
              destinations: (t as any).trip_destinations ?? [],
            }))
          );
        }

    const hasEventItems = selected.some((i) => i.classification === 'Event');

    if (hasEventItems) {
      // Show day-picker to assign all Event items to a single day
      setDayPickerVisible(true);
    } else {
      // All selected items are Tasks — write immediately, no day-picker
      handleWriteTasksOnly(selected);
    }
  }, [items]);

  const handleWriteTasksOnly = async (selected: ExtractionItem[]) => {
    if (!selectedTripId) return;
    setWriting(true);
    try {
      await writeTaskItems(selectedTripId, selected);
      const count = selected.length;
      showToast(`${count} task${count !== 1 ? 's' : ''} added to your checklist`);
      setItems([]);
    } catch {
      Alert.alert('Error', 'Failed to add tasks. Please try again.');
    } finally {
      setWriting(false);
    }
  };

  const handleDaySelected = async (tripDayId: string) => {
    setDayPickerVisible(false);
    if (!selectedTripId) return;

    const selected = items.filter((i) => i.selected);
    setWriting(true);
    try {
      // Write tasks and events simultaneously
      await Promise.all([
        writeTaskItems(selectedTripId, selected),
        writeEventItems(selectedTripId, tripDayId, selected),
      ]);
      const taskCount = selected.filter((i) => i.classification === 'Task').length;
      const eventCount = selected.filter((i) => i.classification === 'Event').length;
      const parts: string[] = [];
      if (eventCount > 0) parts.push(`${eventCount} event${eventCount !== 1 ? 's' : ''}`);
      if (taskCount > 0) parts.push(`${taskCount} task${taskCount !== 1 ? 's' : ''}`);
      showToast(`Added ${parts.join(' and ')}`);
      setItems([]);
    } catch {
      Alert.alert('Error', 'Failed to add items. Please try again.');
    } finally {
      setWriting(false);
    }
  };

  const showToast = (message: string) => {
    setToastMessage(message);
    setTimeout(() => setToastMessage(null), 3000);
  };

  const isLoading = !!pendingJobId && (job.status === 'pending' || job.status === 'processing');

  // No trips case
  if (!tripsLoading && trips.length === 0) {
    return (
      <View style={styles.emptyState}>
        <Text style={styles.emptyTitle}>No trips yet</Text>
        <Text style={styles.emptySubtitle}>Create a trip first to use URL extraction.</Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <TripSelector
        trips={trips}
        selectedTripId={selectedTripId}
        onSelect={setSelectedTripId}
      />
      <UrlInputBar onSubmit={handleUrlSubmit} disabled={isLoading || writing || !selectedTripId} />

      {isLoading && (
        <View style={styles.loadingState}>
          <ActivityIndicator size="large" color="#007AFF" />
          <Text style={styles.loadingText}>Extracting recommendations...</Text>
        </View>
      )}

      {items.length > 0 && !isLoading && (
        <>
          <ExtractionResultsList
            items={items}
            onToggleSelected={handleToggleSelected}
            onToggleClassification={handleToggleClassification}
          />
          <QuickAddButton
            selectedItems={items}
            onQuickAdd={handleQuickAddTap}
            disabled={writing}
          />
        </>
      )}

      {toastMessage && (
        <View style={styles.toast}>
          <Text style={styles.toastText}>{toastMessage}</Text>
        </View>
      )}

      {selectedTripId && (
        <DayPickerSheet
          tripId={selectedTripId}
          visible={dayPickerVisible}
          onDaySelected={handleDaySelected}
          onCancel={() => setDayPickerVisible(false)}
        />
      )}
    </View>
  );
}

import { ExtractionItem } from '@/types/explore';

// Replicate the day-picker trigger logic from EnhanceMyTripScreen for isolated testing
function shouldShowDayPicker(items: ExtractionItem[]): boolean {
  const selected = items.filter((i) => i.selected);
  return selected.some((i) => i.classification === 'Event');
}

function toggleClassification(items: ExtractionItem[], id: string): ExtractionItem[] {
  return items.map((item) =>
    item.id === id
      ? { ...item, classification: item.classification === 'Event' ? 'Task' : 'Event' }
      : item
  );
}

function makeItem(id: string, classification: 'Event' | 'Task', selected = true): ExtractionItem {
  return {
    id,
    recommendation: `Item ${id}`,
    sourceTimestamp: null,
    classification,
    originalClassification: classification,
    selected,
  };
}

describe('Quick Add classification toggle and day-picker trigger', () => {
  test('day-picker shown when at least one selected item is Event', () => {
    const items = [makeItem('1', 'Event'), makeItem('2', 'Task')];
    expect(shouldShowDayPicker(items)).toBe(true);
  });

  test('day-picker not shown when all selected items are Task', () => {
    const items = [makeItem('1', 'Task'), makeItem('2', 'Task')];
    expect(shouldShowDayPicker(items)).toBe(false);
  });

  test('toggling Event to Task removes need for day-picker when no other Events selected', () => {
    let items = [makeItem('1', 'Event'), makeItem('2', 'Task')];

    expect(shouldShowDayPicker(items)).toBe(true);

    expect(items[0].classification).toBe('Event');
    expect(shouldShowDayPicker(items)).toBe(true);
  });

  test('unselected items do not affect day-picker trigger', () => {
    // An Event item that is NOT selected should not trigger day-picker
    const items = [makeItem('1', 'Event', false), makeItem('2', 'Task', true)];
    expect(shouldShowDayPicker(items)).toBe(false);
  });

  test('selecting all Task items — no day-picker; then toggling one to Event — day-picker appears at NEXT quick-add tap', () => {

    expect(shouldShowDayPicker(items)).toBe(true);
  });

  test('empty selection does not trigger day-picker', () => {
    const items = [makeItem('1', 'Event', false), makeItem('2', 'Task', false)];
    expect(shouldShowDayPicker(items)).toBe(false);
  });
